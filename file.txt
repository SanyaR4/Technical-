void add_customer(char file[])
{
    bank customer;
    FILE *fptr = fopen(file,"a");
    if(fptr == NULL)
    {
        printf("File creation error!");
        return;
    }
    customer = getDetails();
    fwrite(&customer,sizeof(customer),1,fptr);
    fclose(fptr);
}
void view_customer(char fileName[])
{
    FILE *fptr = fopen(fileName,"r"); 
    bank customer;
     if(fptr == NULL)
    {
        printf("%s doesnot exist",fileName);
        return;
    }
    while(1)
    {
       fread(&customer,sizeof(customer),1,fptr); 
       if(feof(fptr)) break;
       printDetails(customer);
    }
    fclose(fptr);
}
bank *find_customer(char file[],int accno)
{
    FILE *fptr = fopen(file,"r");
    bank *customer = (bank*)malloc(sizeof(bank));
    while(1)
    {
      fread(&customer,sizeof(customer),1,fptr); 
       if(feof(fptr)) break;
       if(customer->accno == accno)
       {
           return customer;
       }
    }
   fclose(fptr);
   return NULL;
}
